\chapter*{Seznam použitého softwaru}
\addcontentsline{toc}{chapter}{Seznam použitého softwaru}
\begin{itemize}[leftmargin=*]

  % --- Editor a správa projektu ---
  \item \textbf{Visual Studio Code}
        \textit{(rozšíření: LaTeX Workshop, Remote~--~WSL)}
        -- editor zdrojového kódu a správa projektu

  % --- LaTeX distribuce ---
  \item \textbf{TeX Live 2023}
        \textit{(pdflatex, latexmk, biber)}
        -- sazba dokumentu

  % --- LaTeX balíčky (součást TeX Live) ---
  \item \textbf{babel} \textit{(czech, english)} -- jazyková podpora
  \item \textbf{biblatex} \textit{(styl: iso-numeric, backend: biber)} -- citace dle ISO~690
  \item \textbf{TikZ} \textit{(knihovna: external)} -- vektorová grafika a diagramy
  \item \textbf{pgfplots} -- grafy
  \item \textbf{siunitx} -- sazba fyzikálních jednotek
  \item \textbf{acro} -- správa zkratek
  \item \textbf{hyperref} -- hypertextové odkazy v~PDF
  \item \textbf{csquotes} -- typograficky správné uvozovky
  \item \textbf{geometry} -- nastavení stránkových okrajů
  \item \textbf{titlesec} \textit{(pagestyles)} -- formátování nadpisů a záhlaví
  \item \textbf{tocloft} -- formátování obsahu
  \item \textbf{xltabular} -- tabulky přes více stran
  \item \textbf{booktabs} -- profesionální linky tabulek
  \item \textbf{caption}, \textbf{subcaption} -- formátování popisků obrázků
  \item \textbf{pdfpages} -- vkládání stránek PDF
  \item \textbf{microtype} -- mikrotypografické optimalizace
  \item \textbf{ocg-p} -- volitelné vrstvy v~PDF
  \item \textbf{amsmath}, \textbf{amsfonts}, \textbf{amssymb} -- matematická sazba
  \item \textbf{lmodern} -- Latin Modern fonty
  \item \textbf{xcolor} -- definice barev
  \item \textbf{blox} -- bloková schémata (nadstavba TikZ)
  \item \textbf{standalone} -- kompilace dílčích TikZ souborů
  \item \textbf{enumitem}, \textbf{multicol}, \textbf{multirow} -- rozšíření prostředí výčtů a tabulek

  % --- Správa bibliografie ---
  \item \textbf{Zotero} -- správa bibliografických záznamů

  % --- Operační prostředí ---
  \item \textbf{WSL~2 / Ubuntu~22.04} -- prostředí pro překlad v~systému Windows
  \item \textbf{Git} -- správa verzí

  % Python -- doplnit po dokončení sub-projektu python/
\end{itemize}
