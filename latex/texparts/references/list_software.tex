% Inner-list style shared by every software entry:
%   - no label, indented 1.5em
%   - no extra spacing between sub-items
%   - after= removes the outer topsep so sub-list sits tight against the
%     next top-level entry
\newcommand*{\swsubitems}[1]{%
  \begin{itemize}[%
    label={},%
    leftmargin=1.5em,%
    labelsep=0pt,%
    itemsep=0pt,%
    parsep=0pt,%
    topsep=2pt,%
    partopsep=0pt,%
    after=\vspace{0pt}]%
  #1%
  \end{itemize}}
