% ---------------------------------------------------------------------------
%  Czech declension support (acro endings)
% ---------------------------------------------------------------------------
% Czech has 7 grammatical cases.  We define custom endings for the 4 most
% common oblique cases needed in economic/legal prose.  The default ending
% is empty because each acronym sets its own full form via the -form property.
%
% Usage:  \acgen{EU}   → "Evropské unie"  (genitive)
%         \acdat{EU}   → "Evropské unii"  (dative)
%         \acloc{EU}   → "Evropské unii"  (locative)
%         \acacc{EU}   → "Evropskou unii" (accusative)
%         \acins{EU}   → "Evropskou unií" (instrumental)
%
% The short form is always the same abbreviation; only the long form changes.
% On first use: "long-<case>-form (SHORT)" — subsequent: just "SHORT".
%
% Force-first variants (reset + case):
%         \acfacc{APZ} → "aktivní politiku zaměstnanosti (APZ)" regardless of prior usage
%         \acfgen, \acfdat, \acfloc, \acfins — same for other cases
%         \acf{KEY}    → nominative force-first (built-in acro command)
%
% Alt form (built-in acro \aca):
%         \aca{geo-CZ} → "ČR"   (alt= field; valid in Czech prose)
%         \acl{geo-CZ} → "Česko" (full long form; for formal/first mentions)
%         \acs{geo-CZ} → "CZ"   (ISO short; figures and tables ONLY)
% ---------------------------------------------------------------------------
\DeclareAcroEnding{genitive}{}{}
\DeclareAcroEnding{dative}{}{}
\DeclareAcroEnding{accusative}{}{}
\DeclareAcroEnding{locative}{}{}
\DeclareAcroEnding{instrumental}{}{}

% Commands for each case — first occurrence shows "long-case (SHORT)"
\NewAcroCommand\acgen{m}{\acrogenitive\UseAcroTemplate{first}{#1}}
\NewAcroCommand\acdat{m}{\acrodative\UseAcroTemplate{first}{#1}}
\NewAcroCommand\acacc{m}{\acroaccusative\UseAcroTemplate{first}{#1}}
\NewAcroCommand\acloc{m}{\acrolocative\UseAcroTemplate{first}{#1}}
\NewAcroCommand\acins{m}{\acroinstrumental\UseAcroTemplate{first}{#1}}

% Capitalized first-use variants (sentence start).
\NewAcroCommand\Acgen{m}{\acrogenitive\UseAcroTemplate{First}{#1}}
\NewAcroCommand\Acdat{m}{\acrodative\UseAcroTemplate{First}{#1}}
\NewAcroCommand\Acacc{m}{\acroaccusative\UseAcroTemplate{First}{#1}}
\NewAcroCommand\Acloc{m}{\acrolocative\UseAcroTemplate{First}{#1}}
\NewAcroCommand\Acins{m}{\acroinstrumental\UseAcroTemplate{First}{#1}}

% Force-first variants — reset used-flag then show "long-case (SHORT)" unconditionally.
% Use at the start of a paragraph/block where the acronym will repeat frequently and
% needs re-introducing in full.  Nominative force-first = built-in \acf{KEY}.
%   \acfgen{KEY}  → resets + genitive first-form
%   \acfdat{KEY}  → resets + dative first-form
%   \acfacc{KEY}  → resets + accusative first-form   ← most common in Czech prose
%   \acfloc{KEY}  → resets + locative first-form
%   \acfins{KEY}  → resets + instrumental first-form
\NewDocumentCommand\acfgen{m}{\acreset{#1}\acgen{#1}}

%

%   \aclpgen{KEY}  → "odborových organizací"   (gen pl of OO)

%   \aclpacc{KEY}  → "odborové organizace"      (acc pl)
%   \aclploc{KEY}  → "odborových organizacích" (loc pl)
%   \aclpins{KEY}  → "odborovými organizacemi" (ins pl)
%
% First-use commands (long+short on first use, short thereafter):
%   \acpgen{KEY}, \acpdat{KEY}, \acpacc{KEY}, \acploc{KEY}, \acpins{KEY}
% ---------------------------------------------------------------------------
