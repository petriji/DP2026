% ==============================================================================
% Acronym Package Setup
% ==============================================================================
% This file configures the acro package. It must be \input after
%   \usepackage{ocg-p} and \usepackage{acro} in the preamble.
% Acronym definitions (\DeclareAcronym) are in acro.tex.
% ==============================================================================

% ------------------------------------------------------------------------------
% Global options
% ------------------------------------------------------------------------------
% make-links: each \ac{} call becomes a hyperlink to the acronym list entry.
% list/display=used: \printacronyms omits acronyms that never appeared in text.
% list/template=xltabular: use the custom two-column xltabular template below.
% list/heading=chapter*: the printed list gets an unnumbered chapter heading.
% list/preamble: the ocg-p back-link is applied by CTUthesis.cls in
%   \AtBeginDocument via \CTU@acrobacklink (Czech/English localised).
\acsetup{
	make-links     = true,
	list/display   = used,
	list/template  = xltabular,
	list/heading   = chapter*
}

% ------------------------------------------------------------------------------
% List template: xltabular (two-column: bold short | long + foreign + page refs)
% ------------------------------------------------------------------------------
% Column layout: short acronym (bold, optional /alt variant) | long form
%   with optional (foreign, extra) parenthetical, then dot-fill and page refs.
% \AcroRerun: if the list changes between runs, trigger another compilation.
\RenewAcroTemplate[list]{xltabular}{
	\AcroNeedPackage{xltabular}
	\acronymsmapF{%
		\AcroAddRow{
			\textbf{\acrowrite{short}}%
			\acroifT{alt}{/\acrowrite{alt}}
			&
			\acrowrite{list}%
			\acroifanyT{foreign,extra}{ (}%
			\acroifT{foreign}{\acrowrite{foreign}\acroifT{extra}{,}}%
			\acroifT{extra}{\acrowrite{extra}}%
			\acroifanyT{foreign,extra}{)}%
			\acropagefill
			\acropages
			{\acrotranslate{page}\nobreakspace}
			{\acrotranslate{pages}\nobreakspace}%
			\\}}
	{\AcroRerun}%
	\acroheading
	\acropreamble
	\par
	\begin{flushleft}
	\begin{xltabular}[l]{\linewidth}{@{} lX @{}}
	\AcronymTable
	\end{xltabular}
	\end{flushleft}
}

% ------------------------------------------------------------------------------
% Czech locale strings for \printacronyms output
% ------------------------------------------------------------------------------
% Overrides the built-in English strings used in the printed list heading and
%   page-reference labels. "s\abbrdot" produces "s." (strana).
\AddAcroTranslations{Czech}{
	list-name  = Seznam použitých zkratek,
	page       = s\abbrdot,
	pages      = s\abbrdot,
	sequens    = a~dále,
	sequentes  = a~dále,
	also       = také,
	and        = a,
	or         = nebo
}

% ------------------------------------------------------------------------------
% First-use template: long-short
% ------------------------------------------------------------------------------
% On first use: "long form (SHORT)", with an optional footnote for the foreign
%   equivalent and \acrogroupcite to attach a citation to the group.
% On subsequent uses: "SHORT" only.
\RenewAcroTemplate{long-short}{%
	\acroiffirstTF{%
		\acrowrite{long}%
		\acspace(%
		\acrowrite{short}%
		\acroifT{foreign}{\footnote{\acrowrite{foreign}}}%
		\acroifT{alt}{ \acrotranslate{or} \acrowrite{alt}}%
		\acrogroupcite)%
	}%
	{\acrowrite{short}}%
}
