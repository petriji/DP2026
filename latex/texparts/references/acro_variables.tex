% ==============================================================================
% Variable And Index Definitions (acro tag = vel)
% ==============================================================================
% The variables list uses the same acro template as the abbreviations list.
% Used-only behaviour works only when acro sees the symbol as used:
%   - in text: use \acs{var-...} or \ac{var-...}
%   - in math written manually: keep the raw symbol and add \acuse{var-...}
%     nearby so the entry appears in the printed variables list.
% ==============================================================================

% ---------------------------------------------------------------------------
%  Custom SI units for economic quantities
%  Defined here so they are available to all \SI{}{} calls in this thesis.
% ---------------------------------------------------------------------------
\DeclareSIUnit\czk{Kč}
\DeclareSIUnit\eur{€}
\DeclareSIUnit\pps{PPS}
\DeclareSIUnit\week{týd.}
\DeclareSIUnit\month{měs.}
\DeclareSIUnit\person{os.}
% Use slash notation for compound units: \SI{12.5}{\pps\per\hour} → PPS/h
\sisetup{per-mode=symbol}

% ---------------------------------------------------------------------------
%  \DeclareVariable macro
% ---------------------------------------------------------------------------
\newcommand\DeclareVariable[3]{%
	\DeclareAcronym{#1}{%
		short = {#2},
		long = {#3},
		first-style = short,
		single-style = short,
		tag = vel
	}}

% ---------------------------------------------------------------------------
%  Index / modifier symbols  (□ = libovolná veličina, klíč = idx-...)
% ---------------------------------------------------------------------------
\DeclareVariable{idx-b}{\ensuremath{\square_b}}{brutto}
\DeclareVariable{idx-n}{\ensuremath{\square_n}}{netto}
\DeclareVariable{idx-m}{\ensuremath{\square_m}}{měsíční hodnota (výchozí jednotka hodinová)}
\DeclareVariable{idx-pc}{\ensuremath{\square_{pc}}}{per capita (děleno $N$)}
\DeclareVariable{idx-PPS}{\ensuremath{\square^{PPS}}}{hodnota v~paritě kupní síly}
\DeclareVariable{idx-M}{\ensuremath{\square_M}}{gender: muži}
\DeclareVariable{idx-F}{\ensuremath{\square_F}}{gender: ženy}
\DeclareVariable{idx-z}{\ensuremath{\square_z}}{zaměstnanec}
\DeclareVariable{idx-Z}{\ensuremath{\square_Z}}{zaměstnavatel}
\DeclareVariable{idx-SP}{\ensuremath{\square_{SP}}}{sociální pojištění}
\DeclareVariable{idx-ZP}{\ensuremath{\square_{ZP}}}{zdravotní pojištění}
\DeclareVariable{idx-SD}{\ensuremath{\square_{SD}}}{starobní důchod}
\DeclareVariable{idx-KS}{\ensuremath{\square_{KS}}}{kolektivní smlouva}
\DeclareVariable{idx-U}{\ensuremath{\square_U}}{odbory (trade union)}
\DeclareVariable{idx-i}{\ensuremath{\square_i}}{pořadové číslo pozorování (Giniho sumy)}
\DeclareVariable{idx-tilde}{\ensuremath{\tilde{\square}}}{medián veličiny}
\DeclareVariable{idx-bar}{\ensuremath{\bar{\square}}}{aritmetický průměr veličiny}
\DeclareVariable{idx-hat}{\ensuremath{\hat{\square}}}{odhad veličiny}

% ---------------------------------------------------------------------------
%  Base variables
% ---------------------------------------------------------------------------
\DeclareVariable{var-w}{\ensuremath{w}}{%
  mzda; $w_b$ = brutto, $w_n$ = netto,
  dolní index $m$ = měsíční (výchozí jednotka hodinová)
  [\SI{}{\czk\per\hour}]}

\DeclareVariable{var-y}{\ensuremath{y}}{%
  příjem; $y_{SD}$ = starobní důchod
  [\SI{}{\czk\per\month}];
  $y_{pc}^{PPS}$ = čistý disponibilní příjem na ob.\ v~PPS
  [\SI{}{\pps}]}

\DeclareVariable{var-Y}{\ensuremath{Y}}{%
  HDP; $Y_{pc}^{PPS}$ = HDP na obyvatele v~PPS
  [\SI{}{\pps}]}

\DeclareVariable{var-M}{\ensuremath{M}}{%
  majetek; $M_i$ = majetek $i$-tého jedince,
  $\bar{M}$ = průměrný majetek
  [\SI{}{\czk}]}

\DeclareVariable{var-c}{\ensuremath{c}}{%
  odvod / příspěvek;
  $c_{SP}$ = soc.\ pojistné celkem,
  $c_{SP,z}$\,/\,$c_{SP,Z}$ = zaměstnanec\,/\,zaměstnavatel,
  $c_{ZP}$ = zdrav.\ pojistné
  [\SI{}{\czk\per\month}]}

\DeclareVariable{var-n}{\ensuremath{n}}{%
  počet (pojmenovaná podskupina);
  $n_U$ = členové odborů,
  $n_{KS}$ = pracovníci pokrytí KS
  [\SI{}{\person}]}

\DeclareVariable{var-tau}{\ensuremath{\tau}}{%
  daňový klín
  [\SI{}{\percent}]}

\DeclareVariable{var-R-NP}{\ensuremath{R_{NP}}}{%
  náhradový poměr $= y_{SD}/\tilde{w}_{n,m}$
  [\SI{}{\percent}]}

\DeclareVariable{var-R-SP}{\ensuremath{R_{SP}}}{%
  poměr důchodu k~odvodům na SP $= y_{SD}/c_{SP}$
  [1]}

\DeclareVariable{var-L}{\ensuremath{L}}{%
  pracovní síla (ekonomicky aktivní)
  [\SI{}{\person}]}

\DeclareVariable{var-r-KS}{\ensuremath{r_{KS}}}{%
  míra pokrytí kolektivními smlouvami $= n_{KS}/L$
  [\SI{}{\percent}]}

\DeclareVariable{var-rho-U}{\ensuremath{\varrho_U}}{%
  hustota odborové organizace $= n_U/L$
  [\SI{}{\percent}]}

\DeclareVariable{var-N}{\ensuremath{N}}{%
  počet obyvatel / pozorování;
  horní mez sum v~Giniho koeficientu
  [\SI{}{\person}]}

\DeclareVariable{var-g-y}{\ensuremath{g_y}}{%
  Giniho koeficient příjmové nerovnosti
  [0–1]}

\DeclareVariable{var-g-M}{\ensuremath{g_M}}{%
  Giniho koeficient nerovnosti majetku
  [0–1]}

\DeclareVariable{var-r-y}{\ensuremath{r_y}}{%
  poměr disponibilního příjmu k~HDP per capita
  $= y_{pc}^{PPS}/Y_{pc}^{PPS}$
  [1]}

\DeclareVariable{var-G-APZ}{\ensuremath{G_{APZ}}}{%
  výdaje na aktivní politiku zaměstnanosti jako podíl HDP
  [\SI{}{\percent}]}
