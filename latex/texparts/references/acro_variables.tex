% ==============================================================================
% Variable And Index Definitions (acro tag = vel)
% ==============================================================================
% The variables list uses the same acro template as the abbreviations list.
% Used-only behaviour works only when acro sees the symbol as used:
%   - in text: use \acs{var-...} or \ac{var-...}
%   - in math written manually: keep the raw symbol and add \acuse{var-...}
%     nearby so the entry appears in the printed variables list.
% ==============================================================================

\newcommand\DeclareVariable[3]{%
	\DeclareAcronym{#1}{%
		short = {#2},
		long = {#3},
		first-style = short,
		single-style = short,
		tag = vel
	}}

% \DeclareIndex{id}{symbol}{index}{long}
%   For subscript/superscript index variables.
%   #2  full symbol shown in the variables list bold column (e.g. \ensuremath{\square_0})
%   #3  index-only form used in text (e.g. \ensuremath{0}); also works for
%       superscripts (top indices) — just pass the superscript content.
%   #4  long description (same format as \DeclareVariable)
\newcommand\DeclareIndex[4]{%
	\DeclareAcronym{#1}{%
		short = {#2},
		index = {#3},
		long = {#4},
		first-style = short,
		single-style = short,
		tag = vel
	}}
