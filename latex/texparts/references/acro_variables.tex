% ==============================================================================
% Variable And Index Definitions
% ==============================================================================
% Two acro tags are used:
%   tag = vel  → variables       (printed by \printacronyms[include=vel])
%   tag = idx  → indices         (printed by \printacronyms[include=idx])
%
% Used-only behaviour works only when acro sees the symbol as used:
%   - in text: use \acs{var-...}, \acs{idx-...}, or \ac{var-...}
%   - in math written manually: keep the raw symbol and add \acuse{var-...}
%     (or \acuse{idx-...}) nearby so the entry appears in the printed list.
%
% Index display behaviour (see CTUthesis.cls short template):
%   - For idx-* entries the `index` property carries the bare in-text token
%     (e.g. z, M, PPS); \acs{idx-z} renders that token, allowing expressions
%     like  \acs{var-N}_{\acs{idx-z},\acs{idx-M}}  →  N_{z,M}.
%   - The `short` field carries the placeholder form (\square_z, \square^{PPS})
%     and is shown in the bold left column of the indices list.
% ==============================================================================

% ---------------------------------------------------------------------------
%  Custom SI units for economic quantities
%  Defined here so they are available to all \SI{}{} calls in this thesis.
% ---------------------------------------------------------------------------
\DeclareSIUnit\czk{Kč}
\DeclareSIUnit\eur{\text{\texteuro}}
\DeclareSIUnit\pps{PPS}
\DeclareSIUnit\week{týd.}
\DeclareSIUnit\month{měs.}
\DeclareSIUnit\rok{rok}
\DeclareSIUnit\person{os.}
\DeclareSIUnit\pp{p.\,b.}
% Use slash notation for compound units: \SI{12.5}{\pps\per\hour} → PPS/h
\sisetup{per-mode=symbol}

% ---------------------------------------------------------------------------
%  \DeclareVariable macro  (tag = vel — printed in the variables list)
% ---------------------------------------------------------------------------
\newcommand\DeclareVariable[3]{%
	\DeclareAcronym{#1}{%
		short = {#2},
		long = {#3},
		first-style = short,
		single-style = short,
		tag = vel
	}}

% ---------------------------------------------------------------------------
%  \DeclareIndex macro  (tag = idx — printed in the indices list)
%
%  #1 = key (e.g. idx-z)
%  #2 = short / list display, with placeholder \square (e.g. \ensuremath{\square_z})
%        — appears in the bold left column of the indices list.
%  #3 = index display, the bare token used in math (e.g. \ensuremath{z})
%        — what \acs{idx-z} renders inline. Allows building expressions like
%          \acs{var-N}_{\acs{idx-z},\acs{idx-M}}  →  N_{z,M}.
%        The class' short template detects the `index` property (see
%        CTUthesis.cls) and prints it instead of the square form.
%  #4 = long description (right column of the indices list).
% ---------------------------------------------------------------------------
\newcommand\DeclareIndex[4]{%
	\DeclareAcronym{#1}{%
		short = {#2},
		index = {#3},
		long  = {#4},
		first-style = short,
		single-style = short,
		tag = idx
	}}

% ---------------------------------------------------------------------------
%  Accent helpers — plain math macros (no acro bookkeeping)
%
%  Accents (\hat, \bar, \tilde) are standard math notation and do not need
%  separate entries in the indices list. Inside math mode just write
%  \hat{\beta} / \bar{x} / \tilde{w} directly — the variable letter alone
%  carries the acro semantics, the accent is universal notation.
%
%  The \acidx* wrappers below are kept only for backward compatibility
%  with any commentary that already calls them; they expand to the plain
%  math accent and do nothing else.
% ---------------------------------------------------------------------------
\newcommand\acidxtilde[1]{\ensuremath{\tilde{#1}}}
\newcommand\acidxbar[1]{\ensuremath{\bar{#1}}}
\newcommand\acidxhat[1]{\ensuremath{\hat{#1}}}

% ---------------------------------------------------------------------------
%  Base variables
%
%  Convention: each variable is a single symbol.  Compound symbols are
%  built at the call site by combining a base variable with one or more
%  indices, e.g. \acs{var-r}_{\acs{idx-KS}} → r_{KS}.  Every list entry
%  carries the unit in square brackets at the end of the long form.
% ---------------------------------------------------------------------------

% --- Mzda, příjem, majetek -------------------------------------------------
\DeclareVariable{var-w}{\ensuremath{w}}{%
  mzda (\textit{wage}); kombinovat s~indexy: $b$ brutto, $n$ netto,
  $h$/$w$/$m$ hodinová/týdenní/měsíční, $z$ zaměstnanec
  [\SI{}{\czk\per\hour}, \SI{}{\eur\per\hour}, \SI{}{\pps\per\hour}]}

\DeclareVariable{var-y}{\ensuremath{y}}{%
  příjem (\textit{income}); typicky disponibilní příjem domácnosti;
  s~indexy $pc$ per capita, $eq$ ekvivalizovaný, $SD$ starobní důchod,
  horní index $PPS$ v~paritě kupní síly
  [\SI{}{\czk\per\month}, \SI{}{\pps}]}

\DeclareVariable{var-Y}{\ensuremath{Y}}{%
  hrubý domácí produkt (\acs{HDP}); $Y_{pc}^{PPS}$ \acs{HDP} na obyvatele v~PPS
  [\SI{}{\pps}, \SI{}{\eur}]}

\DeclareVariable{var-M}{\ensuremath{M}}{%
  majetek (\textit{wealth}); $M_i$ majetek $i$-tého jedince,
  $\bar{M}$ aritmetický průměr
  [\SI{}{\czk}]}

% --- Práce, čas, hodiny ----------------------------------------------------
\DeclareVariable{var-H}{\ensuremath{H}}{%
  počet odpracovaných hodin; $H_w$ skutečné týdenní hodiny (Eurostat lfsa\_ewhan2),
  $H_a$ roční pracovní fond
  [\SI{}{\hour}]}

\DeclareVariable{var-T}{\ensuremath{T}}{%
  délka časového úseku (období)
  [rok, \SI{}{\week}, \SI{}{\day}]}

\DeclareVariable{var-A}{\ensuremath{A}}{%
  počet pracovních dní v~kalendářním období;
  $A_z$ roční pracovní dny zaměstnance
  [\SI{}{\day}]}

% --- Odvody, daně ----------------------------------------------------------
\DeclareVariable{var-c}{\ensuremath{c}}{%
  odvod / pojistné příspěvek;
  $c_{SP,z}$\,/\,$c_{SP,Z}$ pojistné na \acs{SP} zaměstnance/zaměstnavatele,
  $c_{ZP}$ pojistné na zdravotní pojištění
  [\SI{}{\czk\per\month}]}

\DeclareVariable{var-tau}{\ensuremath{\tau}}{%
  daňový klín (\textit{tax wedge}); podíl odvodů a~daně z~příjmu na
  celkových osobních nákladech zaměstnavatele
  [\SI{}{\percent}]}

% --- Počty, populace, podskupiny -------------------------------------------
\DeclareVariable{var-n}{\ensuremath{n}}{%
  počet členů pojmenované podskupiny;
  $n_U$ členové odborů, $n_{KS}$ pracovníci pokrytí kolektivní smlouvou
  [\SI{}{\person}]}

\DeclareVariable{var-N}{\ensuremath{N}}{%
  celkový počet (obyvatel, pozorování, zaměstnanců); horní mez sum
  v~Giniho koeficientu, jmenovatel přepočtu \textit{per capita}
  [\SI{}{\person}]}

\DeclareVariable{var-L}{\ensuremath{L}}{%
  pracovní síla (ekonomicky aktivní obyvatelstvo)
  [\SI{}{\person}]}

% --- Kvantily, distribuce --------------------------------------------------
\DeclareVariable{var-D3}{\ensuremath{D_3}}{%
  horní hranice třetího decilu čistého ekvivalizovaného příjmu domácností
  (kvantil distribuce, viz Eurostat ilc\_di01)
  [\SI{}{\pps}]}

% --- Poměry, koeficienty ---------------------------------------------------
\DeclareVariable{var-r}{\ensuremath{r}}{%
  bezrozměrný poměr (\textit{ratio}) nebo Pearsonův koeficient korelace
  $r_{xy}$ na intervalu $[-1,\,1]$;
  $r_{KS}$ míra pokrytí \acsp{KS} ($= n_{KS}/L$),
  $r_{y}$ poměr disponibilního příjmu k~\acs{HDP} per capita ($= y_{pc}^{PPS}/Y_{pc}^{PPS}$)
  [1, \SI{}{\percent}]}

\DeclareVariable{var-rho}{\ensuremath{\varrho}}{%
  hustota podskupiny v~populaci nebo Spearmanův koeficient pořadové korelace;
  $\varrho_U$ hustota odborové organizace ($= n_U/L$)
  [\SI{}{\percent}, 1]}

\DeclareVariable{var-R}{\ensuremath{R}}{%
  poměr typu \textit{replacement ratio};
  $R_{NP}$ náhradový poměr důchodu ($= y_{SD}/\tilde{w}_{n,m}$),
  $R_{SP}$ poměr důchodu k~odvodům na \acs{SP} ($= y_{SD}/c_{SP}$)
  [\SI{}{\percent}, 1]}

\DeclareVariable{var-g}{\ensuremath{g}}{%
  Giniho koeficient nerovnosti; $g_y$ příjmové, $g_M$ majetkové
  [0--1]}

\DeclareVariable{var-G}{\ensuremath{G}}{%
  vládní výdaje vyjádřené jako podíl \acs{HDP};
  $G_{APZ}$ výdaje na aktivní politiku zaměstnanosti
  [\SI{}{\percent}]}

% --- Statistický aparát (regrese, korelace) --------------------------------
\DeclareVariable{var-x}{\ensuremath{x}}{%
  vstupní (vysvětlující, nezávislá) proměnná regrese
  [obecná jednotka]}

\DeclareVariable{var-yhat}{\ensuremath{\hat{y}}}{%
  odhad výstupní proměnné z~regresního modelu (predikce)
  [obecná jednotka]}

\DeclareVariable{var-alpha}{\ensuremath{\alpha}}{%
  konstanta (\textit{intercept}) prosté lineární regrese;
  $\hat{\alpha}$ bodový odhad metodou nejmenších čtverců
  [obecná jednotka]}

\DeclareVariable{var-beta}{\ensuremath{\beta}}{%
  sklonový koeficient (\textit{slope}) prosté lineární regrese;
  $\hat{\beta}$ bodový odhad metodou nejmenších čtverců
  [obecná jednotka]}

\DeclareVariable{var-Rdet}{\ensuremath{R^{2}}}{%
  koeficient determinace; podíl rozptylu vysvětlený regresním modelem;
  pro prostou regresi platí $R^2 = r_{xy}^{2}$
  [0--1]}
