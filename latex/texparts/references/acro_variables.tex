% ==============================================================================
% Variable And Index Definitions
% ==============================================================================
% Two acro tags are used:
%   tag = vel  → variables       (printed by \printacronyms[include=vel])
%   tag = idx  → indices         (printed by \printacronyms[include=idx])
%
% Used-only behaviour works only when acro sees the symbol as used:
%   - in text: use \acs{var-...}, \acs{idx-...}, or \ac{var-...}
%   - in math written manually: keep the raw symbol and add \acuse{var-...}
%     (or \acuse{idx-...}) nearby so the entry appears in the printed list.
%
% Index display behaviour (see CTUthesis.cls short template):
%   - For idx-* entries the `index` property carries the bare in-text token
%     (e.g. z, M, PPS); \acs{idx-z} renders that token, allowing expressions
%     like  \acs{var-N}_{\acs{idx-z},\acs{idx-M}}  →  N_{z,M}.
%   - The `short` field carries the placeholder form (\square_z, \square^{PPS})
%     and is shown in the bold left column of the indices list.
% ==============================================================================

% ---------------------------------------------------------------------------
%  \DeclareVariable macro  (tag = vel — printed in the variables list)
% ---------------------------------------------------------------------------
\newcommand\DeclareVariable[3]{%
	\DeclareAcronym{#1}{%
		short = {#2},
		long = {#3},
		first-style = short,
		single-style = short,
		tag = vel
	}}

% ---------------------------------------------------------------------------
%  \DeclareIndex macro  (tag = idx — printed in the indices list)
%
%  #1 = key (e.g. idx-z)
%  #2 = short / list display, with placeholder \square (e.g. \ensuremath{\square_z})
%        — appears in the bold left column of the indices list.
%  #3 = index display, the bare token used in math (e.g. \ensuremath{z})
%        — what \acs{idx-z} renders inline. Allows building expressions like
%          \acs{var-N}_{\acs{idx-z},\acs{idx-M}}  →  N_{z,M}.
%        The class' short template detects the `index` property (see
%        CTUthesis.cls) and prints it instead of the square form.
%  #4 = long description (right column of the indices list).
% ---------------------------------------------------------------------------
\newcommand\DeclareIndex[4]{%
	\DeclareAcronym{#1}{%
		short = {#2},
		index = {#3},
		long  = {#4},
		first-style = short,
		single-style = short,
		tag = idx
	}}

% ---------------------------------------------------------------------------
%  Subscript / superscript indices
%  (\square = libovolná veličina; key = idx-...)

%  In the indices list:  \square_z   zaměstnanec
% ---------------------------------------------------------------------------
\DeclareIndex{idx-b}{\ensuremath{\square_b}}{\ensuremath{b}}{brutto}
\DeclareIndex{idx-n}{\ensuremath{\square_n}}{\ensuremath{n}}{netto}
\DeclareIndex{idx-m}{\ensuremath{\square_m}}{\ensuremath{m}}{měsíční hodnota}
\DeclareIndex{idx-pc}{\ensuremath{\square_{pc}}}{\ensuremath{pc}}{per capita (děleno $N$)}
\DeclareIndex{idx-PPS}{\ensuremath{\square^{PPS}}}{\ensuremath{PPS}}{hodnota v~paritě kupní síly}
\DeclareIndex{idx-M}{\ensuremath{\square_M}}{\ensuremath{M}}{gender: muži}
\DeclareIndex{idx-F}{\ensuremath{\square_F}}{\ensuremath{F}}{gender: ženy}
\DeclareIndex{idx-z}{\ensuremath{\square_z}}{\ensuremath{z}}{zaměstnanec}
\DeclareIndex{idx-Z}{\ensuremath{\square_Z}}{\ensuremath{Z}}{zaměstnavatel}
\DeclareIndex{idx-SP}{\ensuremath{\square_{SP}}}{\ensuremath{SP}}{sociální pojištění}
\DeclareIndex{idx-ZP}{\ensuremath{\square_{ZP}}}{\ensuremath{ZP}}{zdravotní pojištění}
\DeclareIndex{idx-SD}{\ensuremath{\square_{SD}}}{\ensuremath{SD}}{starobní důchod}
\DeclareIndex{idx-KS}{\ensuremath{\square_{KS}}}{\ensuremath{KS}}{kolektivní smlouva}
\DeclareIndex{idx-U}{\ensuremath{\square_U}}{\ensuremath{U}}{odbory (trade union)}

% ---------------------------------------------------------------------------
%  Accent indices — argument-taking macros
%
%  Accents apply to a variable, so they cannot be expressed via the `index`
%  property (which renders an inert symbol). Use the helper macros below
%  whenever you would otherwise write \tilde{X} / \bar{X} / \hat{X}; they
%  register usage so the entry appears in the indices list.
%
%      \acidxtilde{\acs{var-w}}   →   \tilde{w}      (medián)
%      \acidxbar{\acs{var-M}}     →   \bar{M}        (aritmetický průměr)
%      \acidxhat{\acs{var-y}}     →   \hat{y}        (odhad)
%
%  The list entry shows the placeholder form (\tilde{\square} etc.).
% ---------------------------------------------------------------------------
\DeclareIndex{idx-tilde}{\ensuremath{\tilde{\square}}}{\ensuremath{\tilde{\square}}}{medián veličiny}
\DeclareIndex{idx-bar}{\ensuremath{\bar{\square}}}{\ensuremath{\bar{\square}}}{aritmetický průměr veličiny}
\DeclareIndex{idx-hat}{\ensuremath{\hat{\square}}}{\ensuremath{\hat{\square}}}{odhad veličiny}

\newcommand\acidxtilde[1]{\acuse{idx-tilde}\ensuremath{\tilde{#1}}}
\newcommand\acidxbar[1]{\acuse{idx-bar}\ensuremath{\bar{#1}}}
\newcommand\acidxhat[1]{\acuse{idx-hat}\ensuremath{\hat{#1}}}

% ---------------------------------------------------------------------------
%  Base variables
% ---------------------------------------------------------------------------
\DeclareVariable{var-w}{\ensuremath{w}}{%
  mzda; $w_b$ = brutto, $w_n$ = netto,
  dolní index $m$ = měsíční (výchozí jednotka hodinová)
  [\SI{}{\czk\per\hour}]}

\DeclareVariable{var-y}{\ensuremath{y}}{%
  příjem; $y_{SD}$ = starobní důchod
  [\SI{}{\czk\per\month}];
  $y_{pc}^{PPS}$ = čistý disponibilní příjem na ob.\ v~PPS
  [\SI{}{\pps}]}

\DeclareVariable{var-Y}{\ensuremath{Y}}{%
  HDP; $Y_{pc}^{PPS}$ = HDP na obyvatele v~PPS
  [\SI{}{\pps}]}

\DeclareVariable{var-M}{\ensuremath{M}}{%
  majetek; $M_i$ = majetek $i$-tého jedince,
  $\bar{M}$ = průměrný majetek
  [\SI{}{\czk}]}

\DeclareVariable{var-c}{\ensuremath{c}}{%
  odvod / příspěvek;
  $c_{SP}$ = soc.\ pojistné celkem,
  $c_{SP,z}$\,/\,$c_{SP,Z}$ = zaměstnanec\,/\,zaměstnavatel,
  $c_{ZP}$ = zdrav.\ pojistné
  [\SI{}{\czk\per\month}]}

\DeclareVariable{var-n}{\ensuremath{n}}{%
  počet (pojmenovaná podskupina);
  $n_U$ = členové odborů,
  $n_{KS}$ = pracovníci pokrytí KS
  [\SI{}{\person}]}

\DeclareVariable{var-tau}{\ensuremath{\tau}}{%
  daňový klín
  [\SI{}{\percent}]}

\DeclareVariable{var-R-NP}{\ensuremath{R_{NP}}}{%
  náhradový poměr $= y_{SD}/\tilde{w}_{n,m}$
  [\SI{}{\percent}]}

\DeclareVariable{var-R-SP}{\ensuremath{R_{SP}}}{%
  poměr důchodu k~odvodům na SP $= y_{SD}/c_{SP}$
  [1]}

\DeclareVariable{var-L}{\ensuremath{L}}{%
  pracovní síla (ekonomicky aktivní)
  [\SI{}{\person}]}

\DeclareVariable{var-r-KS}{\ensuremath{r_{KS}}}{%
  míra pokrytí kolektivními smlouvami $= n_{KS}/L$
  [\SI{}{\percent}]}

\DeclareVariable{var-rho-U}{\ensuremath{\varrho_U}}{%
  hustota odborové organizace $= n_U/L$
  [\SI{}{\percent}]}

\DeclareVariable{var-D3}{\ensuremath{D_3}}{%
  horní hranice třetího decilu čistého ekvivalizovaného příjmu domácností
  [\SI{}{\pps}]}

\DeclareVariable{var-N}{\ensuremath{N}}{%
  počet obyvatel / pozorování;
  horní mez sum v~Giniho koeficientu
  [\SI{}{\person}]}

\DeclareVariable{var-g-y}{\ensuremath{g_y}}{%
  Giniho koeficient příjmové nerovnosti
  [0–1]}

\DeclareVariable{var-g-M}{\ensuremath{g_M}}{%
  Giniho koeficient nerovnosti majetku
  [0–1]}

\DeclareVariable{var-r-y}{\ensuremath{r_y}}{%
  poměr disponibilního příjmu k~HDP per capita
  $= y_{pc}^{PPS}/Y_{pc}^{PPS}$
  [1]}

\DeclareVariable{var-G-APZ}{\ensuremath{G_{APZ}}}{%
  výdaje na aktivní politiku zaměstnanosti jako podíl HDP
  [\SI{}{\percent}]}
