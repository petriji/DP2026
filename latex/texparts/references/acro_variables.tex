% ==============================================================================
% Variable And Index Definitions (acro tag = vel)
% ==============================================================================
% The variables list uses the same acro template as the abbreviations list.
% Used-only behaviour works only when acro sees the symbol as used:
%   - in text: use \acs{var-...} or \ac{var-...}
%   - in math written manually: keep the raw symbol and add \acuse{var-...}
%     nearby so the entry appears in the printed variables list.
% ==============================================================================

\newcommand\DeclareVariable[3]{%
	\DeclareAcronym{#1}{%
		short = {#2},
		long = {#3},
		first-style = short,
		single-style = short,
		tag = vel
	}}
